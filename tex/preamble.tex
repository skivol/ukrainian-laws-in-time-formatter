\usepackage{pdflscape}
\usepackage{pdfpages}
\usepackage{xtab}
\makeatletter
\def\longtable{\@ifnextchar[\newlongtable@i \newlongtable@ii}
\def\newlongtable@i[#1]{%
\renewcommand{\endhead}{\ignorespaces}
\xtabular[#1]}
\def\newlongtable@ii{%
\renewcommand{\endhead}{\ignorespaces}
\xtabular}
\def\endlongtable{\endxtabular}
\makeatother
